    <script>
        const container = document.getElementById('prompter-container');
        const textElement = document.getElementById('prompter-text');
        const btnToggle = document.getElementById('btn-toggle');
        const btnReset = document.getElementById('btn-reset');
        const btnFlip = document.getElementById('btn-flip');
        const btnRecord = document.getElementById('btn-record');
        const webcamPreview = document.getElementById('webcam-preview');
        const recIndicator = document.getElementById('recording-indicator');
        
        const speedInput = document.getElementById('speed');
        const fontSizeInput = document.getElementById('font-size');

        let isScrolling = false;
        let scrollAnimation;
        let lastTime = 0;

        // Recording State Variables
        let mediaStream = null;
        let mediaRecorder = null;
        let recordedChunks = [];
        let isRecording = false;

        // Initialize user webcam feed on boot
        async function setupCamera() {
            try {
                mediaStream = await navigator.mediaDevices.getUserMedia({
                    video: { width: 1280, height: 720 },
                    audio: true
                });
                webcamPreview.srcObject = mediaStream;
            } catch (err) {
                console.error("Error accessing camera/microphone: ", err);
                alert("Could not access camera or microphone. Please check permissions.");
            }
        }
        setupCamera();

        // Font Adjustments
        fontSizeInput.addEventListener('input', (e) => {
            textElement.style.fontSize = `${e.target.value}px`;
        });

        btnFlip.addEventListener('click', () => {
            textElement.classList.toggle('flipped');
        });

        // Scroll Engine
        function scrollStep(timestamp) {
            if (!isScrolling) return;
            if (!lastTime) lastTime = timestamp;
            const elapsed = timestamp - lastTime;
            const speed = parseFloat(speedInput.value) * 0.05; 
            
            container.scrollTop += speed * elapsed;

            lastTime = timestamp;
            scrollAnimation = requestAnimationFrame(scrollStep);
        }

        function startScrolling() {
            isScrolling = true;
            btnToggle.innerText = "Pause Text";
            lastTime = 0;
            scrollAnimation = requestAnimationFrame(scrollStep);
        }

        function stopScrolling() {
            isScrolling = false;
            btnToggle.innerText = "Play Text";
            cancelAnimationFrame(scrollAnimation);
        }

        function toggleScrolling() {
            if (isScrolling) stopScrolling(); else startScrolling();
        }

        btnToggle.addEventListener('click', toggleScrolling);
        btnReset.addEventListener('click', () => {
            stopScrolling();
            container.scrollTop = 0;
        });

        // Recording System Logic
        btnRecord.addEventListener('click', () => {
            if (!mediaStream) {
                alert("Camera feed not active.");
                return;
            }

            if (!isRecording) {
                // START RECORDING
                recordedChunks = [];
                
                // Use standard formats widely recognized by operating systems (VP8/VP9 or H264 if supported)
                let options = { mimeType: 'video/webm;codecs=vp9,opus' };
                if (!MediaRecorder.isTypeSupported(options.mimeType)) {
                    options = { mimeType: 'video/webm;codecs=vp8,opus' };
                }

                mediaRecorder = new MediaRecorder(mediaStream, options);
                
                mediaRecorder.ondataavailable = (event) => {
                    if (event.data && event.data.size > 0) {
                        recordedChunks.push(event.data);
                    }
                };

                mediaRecorder.onstop = () => {
                    // Turn raw data array buffer into a single downloadable file blob
                    const blob = new Blob(recordedChunks, { type: 'video/webm' });
                    const url = URL.createObjectURL(blob);
                    
                    // Create structural mock element to fire automated download link click
                    const a = document.createElement('a');
                    a.style.display = 'none';
                    a.href = url;
                    a.download = `teleprompter-take-${new Date().toISOString().slice(0,10)}.webm`;
                    document.body.appendChild(a);
                    a.click();
                    
                    setTimeout(() => {
                        document.body.removeChild(a);
                        window.URL.revokeObjectURL(url);
                    }, 100);
                };

                mediaRecorder.start(10); // capture data chunks every 10ms
                
                isRecording = true;
                btnRecord.innerText = "Stop Recording";
                btnRecord.style.backgroundColor = "#333";
                recIndicator.style.display = "block";

                // Auto-trigger prompter context for unified flow
                container.scrollTop = 0;
                startScrolling();

            } else {
                // STOP RECORDING
                mediaRecorder.stop();
                isRecording = false;
                btnRecord.innerText = "Start Recording";
                btnRecord.style.backgroundColor = "var(--record-color)";
                recIndicator.style.display = "none";
                
                stopScrolling();
            }
        });

        // Global hotkey capture
        window.addEventListener('keydown', (e) => {
            if (document.activeElement === textElement) {
                if (e.key === 'Escape') textElement.blur();
                return;
            }
            if (e.key === ' ' || e.code === 'Space') {
                e.preventDefault();
                toggleScrolling();
            }
        });
    </script>
